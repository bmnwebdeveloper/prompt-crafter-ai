import React, { useEffect, useState } from 'react';
import { Layout, Zap, Target, MessageSquare, ShieldCheck, Star, RotateCcw, Copy, Info, ArrowUpRight, AlertTriangle, ExternalLink, Cpu, Instagram, Mail, X, Lock } from 'lucide-react';

// Unique Architect Logo Component (SVG-based for perfect rendering)
const BrandLogo = () => (
  <svg viewBox="0 0 100 100" className="w-full h-full p-1" xmlns="http://www.w3.org/2000/svg">
    <defs>
      <linearGradient id="logoGrad" x1="0%" y1="0%" x2="100%" y2="100%">
        <stop offset="0%" stopColor="#6366f1" />
        <stop offset="100%" stopColor="#a855f7" />
      </linearGradient>
    </defs>
    <path d="M50 10 L85 30 L85 70 L50 90 L15 70 L15 30 Z" fill="none" stroke="url(#logoGrad)" strokeWidth="4" />
    <circle cx="50" cy="50" r="12" fill="url(#logoGrad)" className="animate-pulse" />
    <line x1="50" y1="10" x2="50" y2="38" stroke="url(#logoGrad)" strokeWidth="2" strokeDasharray="2 2" />
    <line x1="15" y1="70" x2="38" y2="58" stroke="url(#logoGrad)" strokeWidth="2" strokeDasharray="2 2" />
    <line x1="85" y1="70" x2="62" y2="58" stroke="url(#logoGrad)" strokeWidth="2" strokeDasharray="2 2" />
  </svg>
);

export default function App() {
  const [input, setInput] = useState('');
  const [isCrafting, setIsCrafting] = useState(false);
  const [hasResult, setHasResult] = useState(false);
  const [copyStatus, setCopyStatus] = useState('Copy Prompt');
  const [selectedStyle, setSelectedStyle] = useState('Professional');
  const [showTerms, setShowTerms] = useState(false);
  const [showPrivacy, setShowPrivacy] = useState(false);

  const contactEmail = "bmnwebdeveloper@gmail.com";

  useEffect(() => {
    try {
      (window.adsbygoogle = window.adsbygoogle || []).push({});
    } catch (e) {
      // Ads might not load in preview mode
    }
  }, [hasResult]);

  const styles = [
    { 
      id: 'Professional', 
      icon: <ShieldCheck className="w-4 h-4" />, 
      desc: 'Formal & Precise',
      longDesc: 'Uses expert-level vocabulary and structured logic. Best for business, technical tasks, or serious academic inquiries.'
    },
    { 
      id: 'Creative', 
      icon: <Star className="w-4 h-4" />, 
      desc: 'Original & Bold',
      longDesc: 'Focuses on unique metaphors, artistic detail, and storytelling. Best for art generation, fiction, or brainstorming new ideas.'
    },
    { 
      id: 'Educational', 
      icon: <Info className="w-4 h-4" />, 
      desc: 'Simple & Clear',
      longDesc: 'Breaks down complex topics into easy-to-understand language. Best for learning new subjects or explaining things to others.'
    },
    { 
      id: 'Persuasive', 
      icon: <Target className="w-4 h-4" />, 
      desc: 'Strong & Convincing',
      longDesc: 'Uses psychological triggers and strong arguments to influence a point of view. Best for debates, marketing, or speeches.'
    }
  ];

  const currentStyleData = styles.find(s => s.id === selectedStyle);

  const processText = () => {
    if (!input.trim()) return;
    setIsCrafting(true);

    setTimeout(() => {
      const userIdea = input.trim();
      const lowerInput = userIdea.toLowerCase();
      let finalPrompt = "";

      const correctionNote = "Note: Please identify and correct any spelling or grammatical errors found in the user's initial idea before generating the result.";
      const executionMandate = "ACTION: Execute the request immediately and provide the full final output. EXPLANATION: After providing the output, provide a concise technical breakdown of your methodology. VARIETY: If this prompt is reused, ensure a unique perspective, different analogies, and a distinct creative approach compared to previous iterations to maintain fresh results.";

      if (lowerInput.includes('code') || lowerInput.includes('app') || lowerInput.includes('dev') || lowerInput.includes('web')) {
        finalPrompt = `As an elite software architect, design a high-performance, modular technical solution for the following idea: "${userIdea}". ${correctionNote} ${executionMandate} Prioritize industry-standard security and clean code practices.`;
      } else if (lowerInput.includes('image') || lowerInput.includes('photo') || lowerInput.includes('visual') || lowerInput.includes('art') || lowerInput.includes('cinematic')) {
        finalPrompt = `A hyper-realistic, cinematic visual masterpiece depicting: "${userIdea}". ${correctionNote} ${executionMandate} Utilize sophisticated 3-point lighting, detailed 8K textures, and a ${selectedStyle.toLowerCase()} artistic direction.`;
      } else {
        finalPrompt = `Construct a profound, expert-level ${selectedStyle.toLowerCase()} analysis regarding: "${userIdea}". ${correctionNote} ${executionMandate} Focus on deep strategic insights and professional clarity.`;
      }

      setInput(finalPrompt);
      setIsCrafting(false);
      setHasResult(true);
    }, 1200);
  };

  const copyToClipboard = () => {
    const el = document.createElement('textarea');
    el.value = input;
    document.body.appendChild(el);
    el.select();
    document.execCommand('copy');
    document.body.removeChild(el);
    setCopyStatus('Copied!');
    setTimeout(() => setCopyStatus('Copy Prompt'), 2000);
  };

  const openSupport = () => {
    window.location.href = `mailto:${contactEmail}`;
  };

  return (
    <div className="min-h-screen bg-slate-950 text-slate-200 font-sans p-4 md:p-8 flex flex-col items-center">
      
      {/* Terms Modal */}
      {showTerms && (
        <div className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-slate-950/80 backdrop-blur-sm">
          <div className="bg-slate-900 border border-slate-800 w-full max-w-2xl rounded-3xl max-h-[80vh] flex flex-col overflow-hidden shadow-2xl animate-in fade-in zoom-in duration-200">
            <div className="p-6 border-b border-slate-800 flex justify-between items-center bg-slate-900/50">
              <h2 className="text-xl font-black tracking-tighter uppercase text-indigo-400">Terms of Service</h2>
              <button onClick={() => setShowTerms(false)} className="p-2 hover:bg-slate-800 rounded-xl transition-colors">
                <X className="w-5 h-5 text-slate-400" />
              </button>
            </div>
            <div className="p-8 overflow-y-auto space-y-6 text-sm text-slate-400 leading-relaxed">
              <section>
                <h3 className="text-slate-100 font-bold mb-2 uppercase text-xs tracking-widest">1. Acceptable Use</h3>
                <p>PromptCrafter AI is designed for professional, creative, and educational purposes. Users must not use the tool to generate content that is harmful, illegal, or violates safety guidelines regarding prohibited substances, violence, or sensitive age-inappropriate material.</p>
              </section>
              <section>
                <h3 className="text-slate-100 font-bold mb-2 uppercase text-xs tracking-widest">2. Intellectual Property</h3>
                <p>The engineering logic, brand identity, and custom SVG assets provided by B.M.N are proprietary. Users retain ownership of their original input ideas, while the engineered prompt output is provided for use in compatible AI systems.</p>
              </section>
              <section>
                <h3 className="text-slate-100 font-bold mb-2 uppercase text-xs tracking-widest">3. User Responsibility</h3>
                <p>Users under 18 must have parental or guardian permission to use this tool. You are responsible for ensuring that the ideas you submit do not violate any local or international laws.</p>
              </section>
              <section>
                <h3 className="text-slate-100 font-bold mb-2 uppercase text-xs tracking-widest">4. Disclaimer</h3>
                <p>PromptCrafter AI provides prompt engineering templates as-is. We do not guarantee the specific quality of third-party AI responses generated from these prompts, as AI outputs are probabilistic by nature.</p>
              </section>
              <section>
                <h3 className="text-slate-100 font-bold mb-2 uppercase text-xs tracking-widest">5. Contact</h3>
                <p>For feedback or legal inquiries, please contact {contactEmail}.</p>
              </section>
            </div>
            <div className="p-6 border-t border-slate-800 bg-slate-900/50 text-center">
              <button onClick={() => setShowTerms(false)} className="px-8 py-2 bg-indigo-600 hover:bg-indigo-500 text-white rounded-xl font-bold transition-all shadow-lg shadow-indigo-500/20">
                I Understand
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Privacy Modal */}
      {showPrivacy && (
        <div className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-slate-950/80 backdrop-blur-sm">
          <div className="bg-slate-900 border border-slate-800 w-full max-w-2xl rounded-3xl max-h-[80vh] flex flex-col overflow-hidden shadow-2xl animate-in fade-in zoom-in duration-200">
            <div className="p-6 border-b border-slate-800 flex justify-between items-center bg-slate-900/50">
              <div className="flex items-center gap-3">
                <Lock className="w-5 h-5 text-indigo-400" />
                <h2 className="text-xl font-black tracking-tighter uppercase text-indigo-400">Privacy Policy</h2>
              </div>
              <button onClick={() => setShowPrivacy(false)} className="p-2 hover:bg-slate-800 rounded-xl transition-colors">
                <X className="w-5 h-5 text-slate-400" />
              </button>
            </div>
            <div className="p-8 overflow-y-auto space-y-6 text-sm text-slate-400 leading-relaxed">
              <p className="italic text-slate-500">Your privacy is extremely important to us. We believe in being transparent about how we handle information.</p>
              
              <section>
                <h3 className="text-slate-100 font-bold mb-2 uppercase text-xs tracking-widest">1. No Personal Data Collection</h3>
                <p>PromptCrafter AI does not require an account, name, or phone number to function. We do not store your input concepts on our servers after your session ends. Your creative ideas remain yours.</p>
              </section>

              <section>
                <h3 className="text-slate-100 font-bold mb-2 uppercase text-xs tracking-widest">2. Usage & Local Storage</h3>
                <p>We may use local session data to help the interface function smoothly. This information is stored only on your device and is not shared with us or any third parties.</p>
              </section>

              <section>
                <h3 className="text-slate-100 font-bold mb-2 uppercase text-xs tracking-widest">3. Third-Party Services</h3>
                <p>We use Google AdSense to show advertisements. Google may use cookies or similar technologies to show ads based on your visits to this and other websites. We do not share your prompt inputs with these advertising partners.</p>
              </section>

              <section>
                <h3 className="text-slate-100 font-bold mb-2 uppercase text-xs tracking-widest">4. Safety for Minors</h3>
                <p>Our tool is designed to be a safe environment for all ages. We do not knowingly collect any data from children. If you are under 18, we encourage you to use this tool responsibly and with adult supervision.</p>
              </section>

              <section>
                <h3 className="text-slate-100 font-bold mb-2 uppercase text-xs tracking-widest">5. Updates</h3>
                <p>This policy may be updated as we add new features. Check back here to stay informed about how we protect your information.</p>
              </section>
            </div>
            <div className="p-6 border-t border-slate-800 bg-slate-900/50 text-center">
              <button onClick={() => setShowPrivacy(false)} className="px-8 py-2 bg-indigo-600 hover:bg-indigo-500 text-white rounded-xl font-bold transition-all shadow-lg shadow-indigo-500/20">
                Close
              </button>
            </div>
          </div>
        </div>
      )}

      {/* TOP AD BANNER */}
      <div className="w-full max-w-6xl mb-6 h-28 bg-slate-900 border border-slate-800 rounded-2xl flex items-center justify-center overflow-hidden relative shadow-lg">
        <div className="absolute top-2 right-3 text-[8px] font-bold text-slate-600 uppercase tracking-widest z-10">Advertisement</div>
        <div className="absolute inset-0 flex flex-col items-center justify-center -z-10">
          <span className="text-indigo-400 font-bold text-sm">Premium AI Partner Space</span>
          <span className="text-[10px] text-slate-500 uppercase">Awaiting Network Approval</span>
        </div>
      </div>

      <div className="max-w-6xl w-full grid grid-cols-1 lg:grid-cols-12 gap-6 items-stretch">
        <div className="lg:col-span-9 space-y-6">
          <header className="flex flex-wrap items-center justify-between gap-4">
            <div className="flex items-center gap-4">
              <div className="w-14 h-14 bg-slate-900 border border-slate-800 rounded-2xl shadow-lg flex items-center justify-center overflow-hidden">
                <BrandLogo />
              </div>
              <div>
                <h1 className="text-2xl font-black tracking-tighter leading-none">
                  PromptCrafter <span className="text-indigo-500">AI</span>
                </h1>
                <p className="text-[10px] font-bold text-slate-500 uppercase tracking-widest mt-1">
                  by B.M.N • Professional Prompt Engineering
                </p>
              </div>
            </div>
            <div className="flex items-center gap-2 bg-slate-900 border border-slate-800 px-4 py-2 rounded-full shadow-inner">
              <div className="w-2 h-2 bg-emerald-500 rounded-full animate-pulse mr-1"></div>
              <span className="text-[10px] font-bold uppercase tracking-widest text-slate-400 whitespace-nowrap">Neural Engine: Optimized</span>
            </div>
          </header>

          <main className={`bg-slate-900 border border-slate-800 rounded-[2.5rem] p-6 transition-all duration-500 ${hasResult ? 'ring-1 ring-indigo-500/50 shadow-2xl shadow-indigo-500/10' : 'shadow-xl'}`}>
            <div className="flex flex-wrap gap-2 mb-6">
              <div className="flex items-center gap-2 px-3 py-1.5 bg-indigo-500/10 border border-indigo-500/20 rounded-lg">
                <ArrowUpRight className="w-3.5 h-3.5 text-indigo-400" />
                <span className="text-[9px] font-black uppercase tracking-widest text-indigo-300">Processing into Pro Templates</span>
              </div>
              <div className="flex items-center gap-2 px-3 py-1.5 bg-amber-500/10 border border-amber-500/20 rounded-lg">
                <AlertTriangle className="w-3.5 h-3.5 text-amber-500" />
                <span className="text-[9px] font-black uppercase tracking-widest text-amber-500">No Auto-Correction Applied</span>
              </div>
            </div>

            <div className="flex justify-between items-center mb-4 px-1">
              <div className="flex items-center gap-2">
                <MessageSquare className="w-4 h-4 text-slate-500" />
                <span className="text-[10px] font-bold uppercase tracking-widest text-slate-500">{hasResult ? 'Constructed Engineering Output' : 'Input Concept Area'}</span>
              </div>
              {hasResult && (
                <div className="flex gap-2">
                  <button onClick={copyToClipboard} className="flex items-center gap-2 px-3 py-1.5 bg-indigo-600 hover:bg-indigo-500 rounded-lg text-xs font-bold transition-all shadow-lg shadow-indigo-500/20"><Copy className="w-3.5 h-3.5" />{copyStatus}</button>
                  <button onClick={() => {setInput(''); setHasResult(false);}} className="flex items-center gap-2 px-3 py-1.5 bg-slate-800 hover:bg-slate-700 rounded-lg text-xs font-bold transition-all"><RotateCcw className="w-3.5 h-3.5" />Reset</button>
                </div>
              )}
            </div>

            <textarea
              className={`w-full h-72 bg-slate-950 border border-slate-800 rounded-3xl p-6 focus:ring-2 focus:ring-indigo-500 outline-none transition-all resize-none font-mono text-base leading-relaxed shadow-inner ${hasResult ? 'text-indigo-100' : 'text-slate-300'}`}
              placeholder="Paste your concept here..."
              value={input}
              onChange={(e) => setInput(e.target.value)}
              disabled={isCrafting}
            />

            {!hasResult && (
              <button
                onClick={processText}
                disabled={isCrafting || !input.trim()}
                className="w-full mt-6 py-6 bg-gradient-to-r from-indigo-700 via-indigo-600 to-violet-700 hover:from-indigo-600 hover:to-violet-600 disabled:opacity-50 text-white font-black text-sm uppercase tracking-[0.2em] rounded-[1.5rem] shadow-2xl shadow-indigo-500/30 flex items-center justify-center gap-3 transition-all active:scale-[0.98] border border-white/10"
              >
                {isCrafting ? (
                  <><div className="w-5 h-5 border-2 border-white/30 border-t-white rounded-full animate-spin"></div><span>Architecting...</span></>
                ) : (
                  <><Cpu className="w-5 h-5 text-indigo-200" /><span>Architect & Refine Elite Prompt</span></>
                )}
              </button>
            )}

            <div className="mt-8 pt-6 border-t border-slate-800/50 flex flex-col gap-6">
              <div className="space-y-4">
                <label className="text-[10px] font-bold text-slate-500 uppercase tracking-widest">Global Output Style:</label>
                <div className="flex flex-wrap gap-2">
                  {styles.map(s => (
                    <button 
                      key={s.id} 
                      onClick={() => setSelectedStyle(s.id)}
                      className={`flex items-center gap-2 px-4 py-2 rounded-xl border transition-all text-[11px] font-bold ${selectedStyle === s.id ? 'bg-indigo-600/10 border-indigo-500 text-indigo-400' : 'bg-slate-950 border-slate-800 text-slate-500 hover:border-slate-700'}`}
                    >
                      {s.icon} {s.id}
                    </button>
                  ))}
                </div>
                {/* Dynamic Style Description Display */}
                <div className="bg-slate-950/80 border border-slate-800/50 p-4 rounded-2xl animate-in fade-in slide-in-from-left-2 duration-300 shadow-inner">
                  <div className="flex items-start gap-3">
                    <div className="mt-1 p-2 bg-indigo-500/10 rounded-lg text-indigo-400">
                      {currentStyleData?.icon}
                    </div>
                    <div>
                      <h4 className="text-[11px] font-black uppercase tracking-widest text-indigo-400 mb-1">
                        {currentStyleData?.id} Mode Active
                      </h4>
                      <p className="text-xs text-slate-400 leading-relaxed max-w-2xl">
                        {currentStyleData?.longDesc}
                      </p>
                    </div>
                  </div>
                </div>
              </div>

              <div className="flex flex-col md:flex-row md:items-center justify-between gap-6">
                <div className="bg-slate-950/50 border border-slate-800 rounded-2xl p-4 flex-1 md:max-w-[200px]">
                   <div className="flex items-center gap-2 mb-1">
                     <Instagram className="w-3 h-3 text-indigo-400" />
                     <p className="text-[10px] text-indigo-400 font-black uppercase tracking-tighter">@b.m.nwe</p>
                   </div>
                   <p className="text-[9px] text-slate-600 leading-tight font-medium uppercase tracking-tighter">Follow for official updates<br /> Designer: B.M.N</p>
                </div>
              </div>
            </div>
          </main>
        </div>

        {/* SIDEBAR AD BANNER */}
        <div className="lg:col-span-3 space-y-4 h-full">
          <div className="bg-slate-900 border border-slate-800 rounded-[2.5rem] p-4 h-full flex flex-col items-center justify-start relative shadow-lg min-h-[400px] overflow-hidden">
            <div className="absolute top-4 right-6 text-[8px] font-bold text-slate-700 uppercase tracking-widest z-10">Sponsored</div>
            <div className="mt-8 w-full flex-grow flex flex-col items-center justify-center p-6 text-center opacity-40">
              <Star className="w-8 h-8 text-indigo-500 mb-4" />
              <h3 className="text-[10px] font-black uppercase tracking-widest text-slate-600">Network Partner</h3>
              <p className="text-[9px] text-slate-700 mt-2">Space Reserved for<br/>Verified AI Partners</p>
            </div>
          </div>
        </div>
      </div>

      <footer className="py-12 text-center w-full max-w-6xl">
        {/* BOTTOM AD BANNER SECTION */}
        <div className="w-full mb-10 h-32 bg-slate-900/50 border border-slate-800 rounded-3xl flex items-center justify-center overflow-hidden relative shadow-lg">
          <div className="absolute top-2 right-4 text-[8px] font-bold text-slate-600 uppercase tracking-widest z-10">Footer Advertisement</div>
          <div className="flex flex-col items-center justify-center text-slate-700">
            <ExternalLink className="w-6 h-6 mb-2 opacity-20" />
            <span className="text-[10px] font-black uppercase tracking-widest">Global Network Marketplace</span>
            <span className="text-[8px] uppercase tracking-tighter mt-1 opacity-50">Secure Ad-Slot Managed by Neural Ad-Engine</span>
          </div>
        </div>

        <div className="flex flex-col items-center gap-4 mb-4">
          <div className="flex justify-center gap-6">
            <button 
              onClick={() => setShowPrivacy(true)}
              className="text-[9px] font-bold text-slate-700 hover:text-slate-400 uppercase tracking-widest transition-colors underline decoration-slate-800 underline-offset-4"
            >
              Privacy
            </button>
            <button 
              onClick={() => setShowTerms(true)}
              className="text-[9px] font-bold text-slate-700 hover:text-slate-400 uppercase tracking-widest transition-colors underline decoration-slate-800 underline-offset-4"
            >
              Terms
            </button>
            <button 
              onClick={openSupport}
              className="text-[9px] font-bold text-slate-700 hover:text-slate-400 uppercase tracking-widest transition-colors underline decoration-slate-800 underline-offset-4"
            >
              Support
            </button>
          </div>
          <div className="flex items-center gap-2 px-4 py-2 bg-slate-900/50 border border-slate-800/50 rounded-xl group cursor-pointer hover:border-indigo-500/30 transition-all" onClick={openSupport}>
            <Mail className="w-3 h-3 text-slate-500 group-hover:text-indigo-400 transition-colors" />
            <span className="text-[10px] font-bold text-slate-500 group-hover:text-slate-300 transition-colors lowercase tracking-wide">
              {contactEmail}
            </span>
          </div>
        </div>
        <p className="text-[9px] font-black text-slate-800 uppercase tracking-[0.5em]">DASHBOARD v6.4 • by B.M.N • @b.m.nwe</p>
      </footer>
    </div>
  );
}
